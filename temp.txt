name: Deploy

on:
  push:
    branches:
      - '**'
  workflow_dispatch:
    inputs:
      environment:
        description: 'Target environment (test|prod)'
        required: true
        default: 'test'

jobs:
  deploy:
    name: 'Deploy'
    runs-on: ubuntu-latest
    needs: build-and-test
    if: github.event_name == 'workflow_dispatch'

    steps:
      - name: Check environment input
        run: |
          if [[ "${{ github.event.inputs.environment }}" != "test" && "${{ github.event.inputs.environment }}" != "prod" ]]; then
            echo "Error: Invalid environment input. Allowed values are 'test' and 'prod'."
            exit 1
          fi

      - name: Check out code
        uses: actions/checkout@v3

      - name: Setup Google Cloud SDK
        uses: google-github-actions/setup-gcloud@v0.2.1
        with:
          service_account_key: ${{ secrets.GCP_SA_KEY }}
          project_id: ${{ secrets.GCP_PROJECT_ID }}

      - name: Install Terraform
        uses: hashicorp/setup-terraform@v2
        with:
          terraform_version: 1.4.6

      - name: Terraform Init and Apply
        run: |
          cd deploy
          terraform workspace select ${{ github.event.inputs.environment }}
          terraform init -backend=false
          terraform fmt -check -recursive
          terraform validate
          terraform apply -auto-approve
          cd ..
